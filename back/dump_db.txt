DROP database IF EXISTS beershop_db;

CREATE database beershop_db;

USE beershop_db;

CREATE TABLE products (
	id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
	name VARCHAR(100) NOT NULL,
	price FLOAT NOT NULL,
	image VARCHAR(150),
	description TEXT
);
